  <script>
    document.addEventListener('DOMContentLoaded', function() {
      // Récupérer les paramètres d'URL
      const urlParams = new URLSearchParams(window.location.search);
      const reference = urlParams.get('ref') || localStorage.getItem('paymentReference');
      const transaction = urlParams.get('transaction') || localStorage.getItem('paymentTransaction');
      
      // Vérifier si nous avons les informations nécessaires
      if (!reference && !transaction) {
        showError('Informations de paiement manquantes. Veuillez vérifier votre e-mail pour les détails de la transaction.');
        return;
      }
      
      // Vérifier le statut du paiement
      if (transaction) {
        verifyPayment(transaction);
      } else {
        // Si nous n'avons que la référence, afficher un message générique
        showConfirmation({
          reference: reference,
          transaction: 'Transaction confirmée',
          date: new Date().toLocaleDateString('fr-FR'),
          amount: '129.80 € TTC',
          status: 'Payé'
        });
      }
    });
    
    // Fonction pour vérifier le paiement
    async function verifyPayment(paymentIntentId) {
      try {
        const response = await fetch(`/api/payment-status/${paymentIntentId}`);
        
        if (!response.ok) {
          throw new Error('Erreur serveur');
        }
        
        const data = await response.json();
        
        if (data.success) {
          // Formater les données pour l'affichage
          const transaction = data.transaction;
          showConfirmation({
            reference: transaction.reference,
            transaction: transaction.id,
            date: new Date(transaction.created).toLocaleDateString('fr-FR'),
            amount: `${transaction.amount.toFixed(2)} ${transaction.currency.toUpperCase()}`,
            status: getStatusText(transaction.status)
          });
          
          // Sauvegarder dans le localStorage
          localStorage.setItem('paymentReference', transaction.reference);
          localStorage.setItem('paymentTransaction', transaction.id);
        } else {
          showError('Paiement non trouvé. Veuillez vérifier votre e-mail pour les détails de la transaction.');
        }
      } catch (error) {
        console.error('Erreur lors de la vérification:', error);
        showError('Une erreur est survenue lors de la vérification du paiement. Veuillez réessayer ultérieurement.');
      }
    }
    
    // Fonction pour afficher la confirmation
    function showConfirmation(details) {
      // Remplir les détails
      document.getElementById('reference').textContent = details.reference;
      document.getElementById('transaction').textContent = details.transaction;
      document.getElementById('date').textContent = details.date;
      document.getElementById('amount').textContent = details.amount;
      document.getElementById('status').textContent = details.status;
      
      // Cacher le chargement et afficher la confirmation
      document.getElementById('loading').style.display = 'none';
      document.getElementById('confirmation').style.display = 'block';
      
      // Configurer le bouton d'impression
      document.getElementById('print-button').addEventListener('click', function(e) {
        e.preventDefault();
        window.print();
      });
    }
    
    // Fonction pour afficher une erreur
    function showError(message) {
      document.getElementById('loading').style.display = 'none';
      document.getElementById('error-message').textContent = message;
      document.getElementById('error').style.display = 'block';
    }
    
    // Fonction pour obtenir le texte du statut
    function getStatusText(status) {
      switch (status) {
        case 'succeeded':
          return 'Payé';
        case 'processing':
          return 'En cours de traitement';
        case 'requires_payment_method':
          return 'Nécessite un moyen de paiement';
        case 'requires_action':
          return 'Nécessite une action';
        default:
          return status.charAt(0).toUpperCase() + status.slice(1);
      }
    }
