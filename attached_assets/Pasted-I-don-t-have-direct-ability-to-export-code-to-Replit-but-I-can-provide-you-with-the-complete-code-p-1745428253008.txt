### server.js (Main backend file)
```javascript
const express = require('express');
const path = require('path');
const bodyParser = require('body-parser');
const cors = require('cors');
require('dotenv').config();

const app = express();
const PORT = process.env.PORT || 3000;

// Middleware
app.use(cors());
app.use(bodyParser.json());
app.use(express.static(path.join(__dirname, 'public')));

// Define routes
app.get('/api/test', (req, res) => {
  console.log('Test API endpoint was called');
  res.json({ message: 'API is working!' });
});

// Form submission route
app.post('/api/submit', async (req, res) => {
  try {
    const {
      name,
      email,
      phone,
      serviceType,
      address,
      city,
      postalCode,
      preferredDate,
      comments
    } = req.body;
    
    // Validate required fields
    if (!name || !email || !phone || !serviceType || !address || !city || !postalCode) {
      return res.status(400).json({ success: false, message: 'All required fields must be provided' });
    }
    
    // Generate reference number
    const referenceNumber = 'REF-' + Math.floor(Math.random() * 10000) + '-' + Date.now().toString().substr(-6);
    
    // Log the submission (in a real app, you would save to database)
    console.log('Form submission received:', {
      ...req.body,
      referenceNumber,
      timestamp: new Date().toISOString()
    });
    
    res.status(201).json({ 
      success: true, 
      message: 'Form submitted successfully',
      referenceNumber: referenceNumber
    });
  } catch (error) {
    console.error('Error submitting form:', error);
    res.status(500).json({ success: false, message: 'An error occurred while processing your request' });
  }
});

// Start server
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
  console.log(`Open https://${process.env.REPL_SLUG}.${process.env.REPL_OWNER}.repl.co in your browser`);
});
```

### public/script.js (Frontend functionality)
```javascript
document.addEventListener('DOMContentLoaded', function() {
    const form = document.getElementById('raccordementForm');
    const statusDiv = document.getElementById('status');
    const submitButton = document.getElementById('submitButton');
    const spinner = document.getElementById('spinner');
    
    // Set minimum date for preferred installation date
    const dateInput = document.getElementById('preferredDate');
    const tomorrow = new Date();
    tomorrow.setDate(tomorrow.getDate() + 1);
    dateInput.min = tomorrow.toISOString().split('T')[0];
    
    // Form validation
    const validateForm = () => {
        let isValid = true;
        
        // Validate name
        const name = document.getElementById('name');
        if (!name.value.trim()) {
            showError(name, 'name-error', 'Veuillez saisir votre nom complet');
            isValid = false;
        } else {
            hideError(name, 'name-error');
        }
        
        // Validate email
        const email = document.getElementById('email');
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!emailRegex.test(email.value)) {
            showError(email, 'email-error', 'Veuillez saisir une adresse email valide');
            isValid = false;
        } else {
            hideError(email, 'email-error');
        }
        
        // Validate phone
        const phone = document.getElementById('phone');
        const phoneRegex = /^\+?[0-9\s\-\(\)]{10,20}$/;
        if (!phoneRegex.test(phone.value)) {
            showError(phone, 'phone-error', 'Veuillez saisir un numéro de téléphone valide');
            isValid = false;
        } else {
            hideError(phone, 'phone-error');
        }
        
        // Validate service type
        const serviceType = document.getElementById('serviceType');
        if (!serviceType.value) {
            showError(serviceType, 'serviceType-error', 'Veuillez sélectionner un type de service');
            isValid = false;
        } else {
            hideError(serviceType, 'serviceType-error');
        }
        
        // Validate address
        const address = document.getElementById('address');
        if (!address.value.trim()) {
            showError(address, 'address-error', 'Veuillez saisir l\'adresse de service');
            isValid = false;
        } else {
            hideError(address, 'address-error');
        }
        
        // Validate city
        const city = document.getElementById('city');
        if (!city.value.trim()) {
            showError(city, 'city-error', 'Veuillez saisir une ville');
            isValid = false;
        } else {
            hideError(city, 'city-error');
        }
        
        // Validate postal code
        const postalCode = document.getElementById('postalCode');
        const postalCodeRegex = /^[A-Za-z0-9\s\-]{3,10}$/;
        if (!postalCodeRegex.test(postalCode.value)) {
            showError(postalCode, 'postalCode-error', 'Veuillez saisir un code postal valide');
            isValid = false;
        } else {
            hideError(postalCode, 'postalCode-error');
        }
        
        return isValid;
    };
    
    const showError = (inputElement, errorId, message) => {
        inputElement.classList.add('input-error');
        const errorElement = document.getElementById(errorId);
        errorElement.textContent = message;
        errorElement.classList.add('show-error');
    };
    
    const hideError = (inputElement, errorId) => {
        inputElement.classList.remove('input-error');
        document.getElementById(errorId).classList.remove('show-error');
    };
    
    // Form submission
    form.addEventListener('submit', function(event) {
        event.preventDefault();
        
        // Validate form
        if (!validateForm()) {
            return;
        }
        
        // Show loading state
        submitButton.disabled = true;
        spinner.style.display = 'inline-block';
        statusDiv.style.display = 'none';
        
        // Collect form data
        const formData = {
            name: document.getElementById('name').value,
            email: document.getElementById('email').value,
            phone: document.getElementById('phone').value,
            serviceType: document.getElementById('serviceType').value,
            address: document.getElementById('address').value,
            city: document.getElementById('city').value,
            postalCode: document.getElementById('postalCode').value,
            preferredDate: document.getElementById('preferredDate').value,
            comments: document.getElementById('comments').value
        };
        
        // Get the current hostname for the API endpoint
        const apiUrl = window.location.origin + '/api/submit';
        
        // Send data to API
        fetch(apiUrl, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(formData)
        })
        .then(response => {
            if (!response.ok) {
                throw new Error('Réponse réseau non valide');
            }
            return response.json();
        })
        .then(data => {
            submitButton.disabled = false;
            spinner.style.display = 'none';
            
            statusDiv.textContent = 'Demande soumise avec succès ! Votre numéro de référence est : ' + data.referenceNumber;
            statusDiv.className = 'success';
            statusDiv.style.display = 'block';
            
            // Reset form
            form.reset();
            
            // Scroll to status message
            statusDiv.scrollIntoView({ behavior: 'smooth' });
        })
        .catch(error => {
            console.error('Erreur:', error);
            
            submitButton.disabled = false;
            spinner.style.display = 'none';
            
            statusDiv.textContent = 'Un problème est survenu lors de la soumission de votre demande. Veuillez réessayer ou contacter le support.';
            statusDiv.className = 'error';
            statusDiv.style.display = 'block';
        });
    });
    
    // Real-time validation as user types
    const inputs = form.querySelectorAll('input, select, textarea');
    inputs.forEach(input => {
        input.addEventListener('blur', function() {
            validateForm();
        });
    });
});
```

### Instructions for Replit Setup:

1. Create a new Node.js Replit project
2. Add all the files above to their respective locations
3. Create a "public" folder and add the frontend files there
4. Run `npm init -y` to initialize package.json (if not already created)
5. Run `npm install express cors body-parser dotenv` to install dependencies
6. Click "Run" to start the server

Once deployed, the form will be accessible at the Replit URL and will be fully functional with mobile-friendly responsive design. The backend will process form submissions and generate reference numbers.
